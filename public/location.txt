	location /static/ {
	  	alias /home/alantelles/Public/;
        	autoindex off; # Deixe como 'on' se quiser listar os arquivos no navegador
        	expires 30d;   # Adiciona cache no navegador por 30 dias (opcional)
		add_header Cache-Control "public, no-transform";

		# --- CONFIGURAÇÃO DO CORS ---

    # 1. Permite qualquer site acessar os arquivos (útil para desenvolvimento local)
    add_header 'Access-Control-Allow-Origin' '*' always;

    # 2. Define os métodos HTTP permitidos
    add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS' always;

    # 3. Define os cabeçalhos permitidos nas requisições do cliente
    add_header 'Access-Control-Allow-Headers' 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization' always;

    # 4. Expõe cabeçalhos para o cliente poder ler respostas longas
    add_header 'Access-Control-Expose-Headers' 'Content-Length,Content-Range' always;

    # 5. Responde instantaneamente às requisições de teste (Preflight/OPTIONS) do navegador
    if ($request_method = 'OPTIONS') {
        add_header 'Access-Control-Allow-Origin' '*' always;
        add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS' always;
        add_header 'Access-Control-Allow-Headers' 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,Authorization' always;
        add_header 'Access-Control-Max-Age' 1728000;
        add_header 'Content-Type' 'text/plain; charset=utf-8';
        add_header 'Content-Length' 0;
        return 204;
    }
	}
